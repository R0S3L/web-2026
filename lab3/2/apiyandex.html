// Visualization Script // Keep existing test pm.test("Status code is 200", function () { pm.response.to.have.status(200); }); // ------------------------------ // Visualizer: Yandex Geocoder // Handles both JSON and non-JSON responses. // If JSON: extracts coordinates and renders scatter chart + small table. // If not JSON: shows error + first 300 chars of body. // ------------------------------ const template = `
Yandex Geocode Visualizer
{{mode}} HTTP {{status}} {{#if query}}Query: {{query}}{{/if}}
{{#if error}}
Could not parse response as JSON
{{error}}
{{bodyPreview}}
{{/if}} {{#unless error}}
Coordinates (scatter)
X = Longitude, Y = Latitude
Extracted points
Top results from response
{{#each points}} {{/each}}
# Name Lat Lon
{{idx}} {{name}} {{lat}} {{lon}}
{{/unless}}
`; function createPayload() { const status = pm.response.code; const bodyText = pm.response.text(); const payload = { mode: 'unknown', status, query: pm.request.url && pm.request.url.query ? (pm.request.url.query.find(q => q.key === 'geocode') || {}).value : undefined, error: null, bodyPreview: null, points: [] }; // Try parse JSON try { const json = pm.response.json(); payload.mode = 'json'; // Yandex geocode: response.GeoObjectCollection.featureMember[].GeoObject const members = json && json.response && json.response.GeoObjectCollection && json.response.GeoObjectCollection.featureMember ? json.response.GeoObjectCollection.featureMember : []; payload.points = members.slice(0, 10).map((m, i) => { const obj = m && m.GeoObject ? m.GeoObject : {}; const name = (obj && obj.name) ? obj.name : ((obj && obj.description) ? obj.description : ('Result ' + (i + 1))); const pos = obj && obj.Point && typeof obj.Point.pos === 'string' ? obj.Point.pos : ''; // pos format: "lon lat" const parts = pos.split(' ').map(p => p.trim()).filter(Boolean); const lon = parts.length > 0 ? Number(parts[0]) : null; const lat = parts.length > 1 ? Number(parts[1]) : null; return { idx: i + 1, name, lat: (typeof lat === 'number' && isFinite(lat)) ? lat : null, lon: (typeof lon === 'number' && isFinite(lon)) ? lon : null }; }).filter(p => p.lat !== null && p.lon !== null); if (!payload.points.length) { payload.mode = 'json (no points found)'; } return payload; } catch (e) { payload.mode = 'non-json'; payload.error = String(e && e.message ? e.message : e); payload.bodyPreview = (bodyText || '').slice(0, 300); return payload; } } pm.visualizer.set(template, createPayload());
